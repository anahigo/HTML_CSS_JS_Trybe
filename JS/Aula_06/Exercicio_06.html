See Solution var img = document.getElementsByTagName('img')[0]; img.style.left = '0px'; var walkForwards = true; function catWalk() { var currentLeft = parseInt(img.style.left); if (walkForwards && (currentLeft > (window.innerWidth-img.width))) { walkForwards = false; } if (!walkForwards && (currentLeft <= 0)) { walkForwards = true; } if (walkForwards) { img.style.left = (currentLeft + 10) + 'px'; } else { img.style.left = (currentLeft - 10) + 'px'; } } window.setInterval(catWalk, 50); Bonus #3: When the cat reaches the middle of the screen, replace the img with an image of a cat dancing, keep it dancing for 10 seconds, and then replace the img with the original image and have it continue the walk.
